13/// Account handlers before the email verification.
14///
15/// This handlers *should be* protected using tokens with
16/// [`TokenScope::EmailVerificationSession`]. See [`SessionHandler`] for
17/// details.
18///
19/// [`SessionHandler`]: super::SessionHandler
20pub struct EmailVerificationSessionHandler(pub AppState);
21
22impl EmailVerificationSessionHandler {
23    /// Resends the sign up email.
24    ///
25    /// Sends a [`ConfirmSignup`] email.
26    ///
27    /// [`ConfirmSignup`]: emails::Template::ConfirmSignup
28    #[tracing::instrument(skip(self))]
29    pub async fn retry_signup(
30        self,
31        id: AccountId,
32        email_dto: dto::request::Email,
33    ) -> HandlerResult<()> {
34        email_dto.validate()?;
35
36        let email = email_dto.email;
37
38        if self.0.account_service.is_email_taken(&email).await? {
39            return Err(ServiceError::EmailTaken)?;
40        }
41
42        let Some(username) = self.0.account_service.get_primary_username(id).await? else {
43            return Err(HandlerError::UnexpectedError("account with no username"));
44        };
45
46        let complete_signup_jwt = self.0.token_service.sign(&Token {
47            id,
48            scope: TokenScope::CompleteSignup {
49                email: email.clone(),
50            },
51        });
52
53        let template = emails::Template::ConfirmSignup {
54            username,
55            token: complete_signup_jwt,
56        };
57
58        self.0.email_client.send(email, id, template).await;
59
60        Ok(())
61    }
62}
