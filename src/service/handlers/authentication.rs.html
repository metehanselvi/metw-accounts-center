11static SIGNUP_TOKEN_VALID_FOR: Duration = Duration::from_secs(60 * 60 * 24);
12static LOGIN_TOKEN_VALID_FOR: Duration = Duration::from_secs(60 * 60 * 24 * 7);
13
14/// Gateway handlers for creating accouts or logging into accounts.
15pub struct AuthenticationHandler(pub State);
16
17impl AuthenticationHandler {
18    /// Verify the authentication token.
19    #[tracing::instrument(skip_all)]
20    pub async fn auth(self, base64_encoded_token: String) -> HandlerResult<AccountId> {
21        let token = self.0.token_service.verify(&base64_encoded_token).await?;
22
23        if let TokenScope::Authenticate = token.scope {
24            Ok(token.id)
25        } else {
26            Err(HandlerError::Unauthorized)
27        }
28    }
29
30    /// POST `/signup`
31    #[tracing::instrument(skip_all, fields(username = signup_dto.username, email = signup_dto.email))]
32    pub async fn signup(self, signup_dto: dto::request::Signup) -> HandlerResult<AccountId> {
33        let email = signup_dto.email.clone();
34        let username = signup_dto.username.clone();
35
36        let account_id = self.0.account_service.signup(signup_dto).await?;
37
38        let signup_jwt = self.0.token_service.sign(&Token::new(
39            account_id,
40            TokenScope::Signup {
41                email: email.clone(),
42            },
43            SIGNUP_TOKEN_VALID_FOR,
44        ));
45
46        let template = mails::Template::Signup {
47            username,
48            signup_jwt,
49            callback_url: self.0.email_callback_url,
50        };
51
52        self.0.mail_client.send(email, account_id, template).await;
53
54        Ok(account_id)
55    }
56
57    /// POST `/login` (with `username`)
58    #[tracing::instrument(skip_all, fields(username = login_dto.username))]
59    pub async fn login_by_username(
60        self,
61        login_dto: dto::request::LoginWithUsername,
62    ) -> HandlerResult<String> {
63        let account_id = self
64            .0
65            .account_service
66            .login_with_username(login_dto)
67            .await?;
68
69        Ok(self.login(account_id))
70    }
71
72    /// POST `/login` (with `email`)
73    #[tracing::instrument(skip_all, fields(email = login_dto.email))]
74    pub async fn login_by_email(
75        self,
76        login_dto: dto::request::LoginWithEmail,
77    ) -> HandlerResult<String> {
78        let account_id = self.0.account_service.login_with_email(login_dto).await?;
79
80        Ok(self.login(account_id))
81    }
82
83    fn login(self, account_id: AccountId) -> String {
84        tracing::trace!(%account_id);
85
86        self.0.token_service.sign(&Token::new(
87            account_id,
88            TokenScope::Authenticate,
89            LOGIN_TOKEN_VALID_FOR,
90        ))
91    }
92}
