1use crate::id::AccountId;
2use serde::{Deserialize, Serialize};
3use std::time::Duration;
4
5/// Authentication token with authorization scopes.
6#[derive(Debug)]
7pub struct Token {
8    /// Account id.
9    pub id: AccountId,
10    /// Token's permissions.
11    pub scope: TokenScope,
12    /// Duration the token is valid for.
13    pub valid_for: Duration,
14}
15
16/// Authorization scopes.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18#[allow(missing_docs)]
19pub enum TokenScope {
20    /// Permit logins.
21    Authenticate,
22
23    /// Permit adding the email to the account.
24    AddEmail(String),
25    /// Allow changing account's primary email address to given address.
26    SetPrimaryEmail {
27        current_primary_email: String,
28        new_primary_email: String,
29    },
30    /// Enable account and add first primary email. This scope is present in
31    /// email sent in signup procedure.
32    Signup { email: String },
33}
34
35impl Token {
36    /// Create a new token.
37    pub fn new(id: AccountId, scope: TokenScope, valid_for: Duration) -> Self {
38        Token {
39            id,
40            scope,
41            valid_for,
42        }
43    }
44}
45
46impl TokenScope {
47    /// Get name of the enum variant.
48    pub fn variant_name(&self) -> &'static str {
49        match self {
50            Self::Authenticate => "authenticate",
51            Self::AddEmail(..) => "add-email",
52            Self::SetPrimaryEmail { .. } => "set-primary-email",
53            Self::Signup { .. } => "signup",
54        }
55    }
56}
