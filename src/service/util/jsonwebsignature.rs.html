12/// JSON web signature (JWS).
13pub struct JsonWebSignature {
14    secret: jws::Secret,
15}
16
17#[derive(Deserialize, Serialize)]
18struct PrivateClaims {
19    scope: TokenScope,
20    id: AccountId,
21}
22
23#[cfg(not(test))]
24static NOW: fn() -> DateTime<Utc> = Utc::now;
25#[cfg(test)]
26static NOW: fn() -> DateTime<Utc> = JsonWebSignature::injected_now;
27
28#[cfg(test)]
29static INJECTED_NOW: Mutex<Option<DateTime<Utc>>> = Mutex::new(None);
30
31impl JsonWebSignature {
32    /// Creates a new JWS verifier/signer for [`Token`].
33    pub fn new(secret: Vec<u8>) -> Self {
34        Self {
35            secret: jws::Secret::Bytes(secret),
36        }
37    }
38
39    /// Sign and encode the token.
40    pub fn encode(&self, token: &Token) -> String {
41        let now = NOW();
42
43        let payload = biscuit::ClaimsSet::<PrivateClaims> {
44            registered: biscuit::RegisteredClaims {
45                expiry: Some((now + token.valid_for).into()),
46                not_before: Some(now.into()),
47                issued_at: Some(now.into()),
48                ..Default::default()
49            },
50            private: PrivateClaims {
51                scope: token.scope.clone(),
52                id: token.id,
53            },
54        };
55        let header = jws::Header::<biscuit::Empty> {
56            registered: jws::RegisteredHeader {
57                algorithm: jwa::SignatureAlgorithm::HS256,
58                ..Default::default()
59            },
60            ..Default::default()
61        };
62
63        let jwt = JWT::new_decoded(header, payload);
64
65        jwt.into_encoded(&self.secret)
66            .unwrap()
67            .encoded()
68            .unwrap()
69            .to_string()
70    }
71
72    /// Decode the token by verifying it.
73    pub fn decode(&self, base64_encoded_token: &str) -> Option<(Token, Vec<u8>)> {
74        let token =
75            biscuit::JWT::<PrivateClaims, biscuit::Empty>::new_encoded(base64_encoded_token);
76
77        let signature = token.signature().ok()?;
78
79        let token = token
80            .into_decoded(&self.secret, jwa::SignatureAlgorithm::HS256)
81            .ok()?;
82
83        let now = NOW();
84
85        token
86            .validate(biscuit::ValidationOptions {
87                claim_presence_options: biscuit::ClaimPresenceOptions {
88                    issued_at: biscuit::Presence::Required,
89                    not_before: biscuit::Presence::Required,
90                    expiry: biscuit::Presence::Required,
91                    ..Default::default()
92                },
93                temporal_options: biscuit::TemporalOptions {
94                    now: Some(NOW()),
95                    ..Default::default()
96                },
97                ..Default::default()
98            })
99            .ok()?;
100
101        let payload = token.payload().unwrap();
102        let expiry = *payload.registered.expiry.unwrap();
103
104        let valid_for = if expiry > now + Duration::seconds(60) {
105            (expiry - now).to_std().unwrap()
106        } else {
107            std::time::Duration::from_secs(60)
108        };
109
110        Some((
111            Token {
112                id: payload.private.id,
113                scope: payload.private.scope.clone(),
114                valid_for,
115            },
116            signature,
117        ))
118    }
119
120    #[cfg(test)]
121    pub(crate) fn inject_now(date_time: Option<DateTime<Utc>>) {
122        *INJECTED_NOW.lock().unwrap() = date_time
123    }
124
125    #[cfg(test)]
126    fn injected_now() -> DateTime<Utc> {
127        if let Some(now) = *INJECTED_NOW.lock().unwrap() {
128            now
129        } else {
130            Utc::now()
131        }
132    }
