3use crate::{

16#[utoipa::path(
17    get, path = "",
18    security(("session_jwt" = [])),
19    responses(
20        (status = OK, body = dto::response::Account)
21    )
22)]
23async fn me(
24    State(state): State<AppState>,
25    Extension(id): Extension<AccountId>,
26) -> AppResult<dto::response::Account> {
27    Ok(AppJson(SessionHandler(state).me(id).await?))
28}
29
30#[utoipa::path(
31    post, path = "/emails",
32    security(("session_jwt" = [])),
33    request_body = dto::request::Email,
34    responses(
35        (status = OK)
36    )
37)]
38async fn add_email(
39    State(state): State<AppState>,
40    Extension(id): Extension<AccountId>,
41    AppJson(email_dto): AppJson<dto::request::Email>,
42) -> AppResult<()> {
43    Ok(AppJson(
44        SessionHandler(state).add_email(id, email_dto).await?,
45    ))
46}
47
48#[utoipa::path(
49    delete, path = "/emails",

51    request_body = dto::request::Email,
52    responses(
53        (status = OK)
54    )
55)]
56async fn delete_email(
57    State(state): State<AppState>,

66#[utoipa::path(
67    post, path = "/emails/set-primary",
68    security(("session_jwt" = [])),
69    request_body = dto::request::Email,
70    responses(
71        (status = OK)
72    )
73)]
74async fn set_primary_email(
75    State(state): State<AppState>,
76    Extension(id): Extension<AccountId>,
77    AppJson(email_dto): AppJson<dto::request::Email>,
78) -> AppResult<()> {
79    Ok(AppJson(
80        SessionHandler(state)
81            .set_primary_email(id, email_dto)
82            .await?,
83    ))
84}
85
86pub fn routes(state: AppState) -> Router {
87    Router::new()
88        .route("/me", get(me))
89        .route("/me/emails", post(add_email))
90        .route("/me/emails", delete(delete_email))
91        .route("/me/emails/set-primary", post(set_primary_email))
92        .route_layer(middleware::from_fn_with_state(
93            state.clone(),
94            crate::middleware::auth_session,
95        ))
96        .with_state(state)
97}
